$('.collapsed').each(function() {
	$(this).height($(this).attr('data-collapsedHeight'));
});


//code for toggling collapse/expand
$('.collapsed>span').click(function() {
	if ($(this).hasClass('collapseBtnd')) { //if is expanded

		//reset to how it was after initialization
		$(this).parent().attr('style', '').height($(this).parent().attr('data-collapsedHeight'));

		$(this).removeClass('collapseBtnd');

	}
	else{
		//otherwise expand div and add a class to flip the arrow around (indicate a collapse option)
		$(this).parent().css({
			'height': 'auto',
			'overflow': 'visible',
			'margin-bottom': parseInt($(this).parent().css('margin-bottom')) + 60 + 'px'
		});
		$(this).addClass('collapseBtnd');
	};
});
