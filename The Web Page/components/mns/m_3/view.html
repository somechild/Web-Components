// for demo purposes: randomly assigns colors to each link item in the list
	// --> you will have to custom enter data-colorI="{1-6}" into every li item based on which color you prefer to use for it
var usedRands = [];
$(document).ready(function() {
	$('#header>div>ul>li').each(function() {
		var rand = (parseInt(Math.random() * 10)%6 + 1);
		while (usedRands.indexOf(rand) !== -1)
			rand = (parseInt(Math.random() * 10)%5 + 1);

		usedRands.push(rand);
		$(this).attr('data-colorI', rand);
	});
});
